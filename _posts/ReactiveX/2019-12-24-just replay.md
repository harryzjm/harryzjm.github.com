```swift  
var num = 1
let signal = Observable<TimeInterval>
    .create { (observe) -> Disposable in
        let current = num; defer { num += 1 }
        print("---------------------------new \(current)")
        observe.onNext(Date().timeIntervalSince1970)
        observe.onNext(Date().timeIntervalSince1970)
        DispatchQueue.after(.seconds(1)) {
            observe.onNext(Date().timeIntervalSince1970)
            observe.onCompleted()
            print("---------------------------done \(current)")
        }
        return Disposables.create() }
//  .replayAll()

print("----------------------1")
signal.bind { (i) in
    print("1", i)
}.disposed(by: bag)

print("----------------------2")
signal.bind { (i) in
    print("2", i)
}.disposed(by: bag)

//signal.connect().disposed(by: bag)

DispatchQueue.after(.milliseconds(500)) {
    print("----------------------3")
    signal.bind { (i) in
        print("3", i)
    }.disposed(by: self.bag)
}

DispatchQueue.after(.seconds(2)) {
    print("----------------------4")
    signal.bind { (i) in
        print("4", i)
    }.disposed(by: self.bag)
}
```

1.有订阅后即开始发送 各自生成自己的信号流

3.信号已结束 跟进订阅: 产生新信号流  

### publish()  
即 `replay(0)`  
```bash  
----------------------1
----------------------2
---------------------------new 1
1 1577239639.52909
2 1577239639.52909
1 1577239639.529395
2 1577239639.529395
----------------------3
1 1577239640.581406
2 1577239640.581406
3 1577239640.581406
---------------------------done 1
----------------------4
```  
1.控制首发 共享  
2.信号未结束 跟进订阅: 同步等待  
3.信号已结束 跟进订阅: 无任何信号流
